func (g *Graph) DFSIterative(startNode int) {
    visited := make(map[int]bool)
    stack := []int{startNode}

    for len(stack) > 0 {
        // Pop the last element testtttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttttt
        node := stack[len(stack)-1]
        stack = stack[:len(stack)-1]

        if !visited[node] {
            fmt.Printf("%d ", node)
            visited[node] = true
        }

        // Push unvisited neighbors to the stack
        for _, neighbor := range g.adj[node] {
            if !visited[neighbor] {
                stack = append(stack, neighbor)
            }
        }
    }
}
